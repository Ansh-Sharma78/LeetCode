/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public void pre(TreeNode temp, List<TreeNode> list){
        if(temp==null) return;

        list.add(temp);
        pre(temp.left, list);
        pre(temp.right, list);
    }
    public void flatten(TreeNode root) {
        List<TreeNode> list=new ArrayList<>();
        if(root==null){
            return;
        }
        pre(root, list);
        TreeNode temp=root;
        if(root.left!=null) root.left=null;
        if(root.right!=null) root.right=null;

        int i=1;
        while(i<list.size()){
            TreeNode t=list.get(i);
            if(t.left!=null) t.left=null;
            if(t.right!=null) t.right=null;

            temp.right=t;
            temp=t;
            i++;
        }
        

    }
}
