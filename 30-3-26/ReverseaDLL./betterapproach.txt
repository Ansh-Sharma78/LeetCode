class Solution {
    public Node reverse(Node head) {
          if(head==null || head.next==null){
            return head; 
        }
        // code here
        Node curr=head;
        Node prev=null;
        Node next;
        
        while(curr!=null){
            next=curr.next;
            curr.next=prev;
            curr.prev=next;
            prev=curr;
            curr=next;
        }
        head=prev;
      
      return head;
    }
}
