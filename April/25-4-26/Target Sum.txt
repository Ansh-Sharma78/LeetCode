class Solution {
    public int rec(int i, int sum, int nums[], int target){
        if(i==nums.length){
           return sum==target ? 1:0;
        }

        sum+=nums[i];
        int add= rec(i+1, sum, nums, target);
        sum-=nums[i];

        sum= sum - nums[i];
        int sub= rec(i+1, sum, nums, target);

        return add+sub;
    }
    public int findTargetSumWays(int[] nums, int target) {

        int sum=0;
        return rec(0, sum, nums, target);
       
      
    }
}
