/*
// Definition for a Node.
class Node {
    int val;
    Node next;
    Node random;

    public Node(int val) {
        this.val = val;
        this.next = null;
        this.random = null;
    }
}
*/

class Solution {
    public Node copyRandomList(Node head) {
     
        Node res;
        Node temp=head;

        HashMap<Node,Node> map=new HashMap<>();
        while(temp!=null){
            Node t= new Node(temp.val);
            map.put(temp,t);
            temp=temp.next;
        }
        temp=head;
        res=map.get(temp);
        while(temp!=null){
            Node l=map.get(temp);
            l.next=map.get(temp.next);
            l.random=map.get(temp.random);
            temp=temp.next;
        }
       

        return res;
    }
}
