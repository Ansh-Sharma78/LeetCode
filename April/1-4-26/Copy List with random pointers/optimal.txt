//WITH NO HASHMAP TC->O(3N) + SC-> O(1) + O(N)-(Required for building ll);
class Solution {
    public Node copyRandomList(Node head) {
        if(head==null ){
            return head;
        }
        Node dummy=new Node(-1);
        Node res=dummy;
       Node temp=head;
       while(temp!=null){
         Node t=new Node(temp.val);
         t.next=temp.next;
         temp.next=t;
         temp=temp.next.next;
       }
       temp=head;

       while(temp!=null){
        if(temp.random==null){
            temp.next.random=null;
            temp=temp.next.next;
            continue;
        }
         temp.next.random=temp.random.next;
         temp=temp.next.next;
       }
       temp=head;
       res=temp.next;

       while(temp!=null){
          dummy.next=temp.next;
          dummy=temp.next;
          temp.next=temp.next.next;
          temp=dummy.next;
       }
       return res;
    }
}
