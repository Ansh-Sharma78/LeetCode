class Solution {
    public void rec(TreeNode temp, List<Integer> res){
        if(temp==null){
            return;
        }
        res.add(temp.val);
        rec(temp.left,res);
        rec(temp.right,res);
    }
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> res= new ArrayList<>();
        rec(root, res);
        return res;
    }
}
