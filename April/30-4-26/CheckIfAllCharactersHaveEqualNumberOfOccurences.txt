class Solution {
    public boolean areOccurrencesEqual(String s) {
        int freq[]=new int[26];

        for(int i=0;i<s.length();i++){
            freq[s.charAt(i)-'a']++;
        }

        int count=freq[s.charAt(0)-'a'];
        for(int i : freq){
            if(i!=count && i!=0) return false;
        }
        return true;
    }
}
