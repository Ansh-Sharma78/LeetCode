/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

 //MY APPROACH 

 
class Solution {
    public void rec(TreeNode temp, List<List<Integer>> res,int level){
       if(temp==null){
        return;
       }
       if(res.size()==level){
        res.add(new ArrayList<>());
       }
       res.get(level).add(temp.val);

       rec(temp.left, res, level+1);
       rec(temp.right,res, level+1);
       
    }
    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> res=new ArrayList<>();
        rec(root, res, 0);
        boolean reverse=false;
        for(List<Integer> t : res){
            if(reverse==true){
                Collections.reverse(t);
            }
            reverse=(!reverse);
        }
        return res;
    }
}

//Approach 2
Using que and iteration

class Solution {
    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
       List<List<Integer>> res = new ArrayList<>();
       if(root==null) return res;
       ArrayDeque<TreeNode> q=new ArrayDeque<>();
       
       q.addLast(root);
       boolean reverse=false;

       while(!q.isEmpty()){
         int k=q.size();
         List<Integer> temp=new ArrayList<>();
         for(int i=0;i<k;i++){
            TreeNode t=q.poll();
            temp.add(t.val);
            
            if(t.left!=null) q.add(t.left);
            if(t.right!=null) q.add(t.right);
         }
         if(reverse) Collections.reverse(temp);
         res.add(temp);
         reverse=(!reverse);
       }
       return res;
       
    }
}
