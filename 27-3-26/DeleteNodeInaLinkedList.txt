/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    // here we just copied the value of the next node of node provided and deleted the next node of the node which has to be deleted,
    //since we havent provided head and tail it is impossible to traverse the list without that
    public void deleteNode(ListNode node) {
        ListNode temp=node;
        temp.val=temp.next.val;
        temp.next=temp.next.next;
    }
}
