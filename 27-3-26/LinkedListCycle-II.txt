//detecting in a linkedlist where a loop is forming and returning its index
/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode detectCycle(ListNode head) {
        if(head==null || head.next==null || head.next.next==null){
            return null;
        }
        ListNode slow=head.next;
        ListNode fast=head.next.next;
       // slow=slow.next;
    
        while(fast!=null && fast.next!=null){
            if(slow==fast){
                break;
            }
            slow=slow.next;
            fast=fast.next.next;
        }
        if(fast==null || fast.next==null){
            return null;
        }
        
        slow=head;
        while(slow!=fast){
            
            slow=slow.next;
            fast=fast.next;
        }
        return slow;


    }
}
